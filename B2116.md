#include<iostream>
#include<string>

using namespace std;

int main ()
{
	string s;
	getline(cin, s);
	int len = s.size();
	for(int i = 0;i < len; i++){
		if(s[i] >= 'a' && s[i] <= 'z'){
			s[i] -= 32;
		}
		else if(s[i] >= 'A' && s[i] <= 'Z'){
			s[i] += 32;
		}
	}
	for(int i = 0; i < len / 2; i++){
		swap(s[i], s[len - i - 1]);
	}
	for(int i = 0; i < len; i++){
		if(s[i] >= 'a' && s[i] <= 'z'){
			s[i] = (s[i] - 'a' + 3) % 26 + 'a';
		}
		else if(s[i] >= 'A' && s[i] <= 'Z'){
			s[i] = (s[i] - 'A' + 3) % 26 + 'A';
		}	
	}
	cout << s << endl;
	return 0;
	
}
