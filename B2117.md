#include<iostream>
#include<string>
#include<cctype>
using namespace std;

int main ()
{
	int num = 0;
	cin >> num;
	for(int i = 0;i < num; i++){
		string s;
		cin >> s;
		int len = s.size();
		for(int i = 0;i < len; i++){
			if(i == 0){
				s[0] = toupper(s[0]);
			}
			else{
				s[i] = tolower(s[i]);
			}
		}
		cout << s << endl;
	}
}
