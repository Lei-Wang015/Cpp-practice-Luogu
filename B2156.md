```
#include<iostream>
#include<string>
using namespace std;
int main ()
{
	string s,ans;
	while(cin >> s){
		bool end = false;
		int len = s.size();
		if(s[len - 1] == '.'){
			s.pop_back();
			end = true;
		}
		if(s.size() > ans.size()){
			ans = s;
		}
		if(end){
			break;
		}
	}
	cout << ans << endl;
	return 0;
}
```
