```
#include<iostream>
#include<cstdlib>
using namespace std;
int main ()
{
	int n = 0;
	cin >> n;
	int p[n], j[n], s[n];
	for(int i = 0;i < n;i++){
		cin >> p[i];
	}
	for(int i = 0;i < n;i++){
		cin >> j[i];
	}
	for(int i = 0;i < n;i++){
		cin >> s[i];
	}
	int score[n];
	int sum = 0;
	for(int i = 0;i < n;i++){
		score[i] = s[i] - (abs(p[i] - j[i]));
		if(score[i] >= 0){
			sum += score[i];
		}
	}
	cout << sum << endl;
	return 0;
}
```
