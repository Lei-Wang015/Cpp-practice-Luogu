#include<iostream>
#include<string>

using namespace std;
int value(string s);
int main ()
{
	int num;
	cin >> num;
	for(int i = 0; i < num; i++){
		string s1,s2;
		cin >> s1 >> s2;
		int val1 = value(s1);
		int val2 = value(s2);
		int diff = val1 - val2;
		if(diff == 1 || diff == -2){
			cout << "Player1" << endl;
		}
		else if(diff == 0){
			cout << "Tie" << endl;
		}
		else{
			cout << "Player2" << endl;
		}
	}
	return 0;
}

int value(string s){
	if(s == "Rock"){
		return 1;
	}
	else if(s == "Scissors"){
		return 0;
	}
	else if(s == "Paper"){
		return -1;
	}
	return 0;
}
