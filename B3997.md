#include<iostream>
#include<string>

using namespace std;

int isPalindrome(string s, int start, int end);

int main ()
{
	string s;
	cin >> s;
	int start_len = 0;
	int current_len = 0;
	int cnt = 0;
	int len = s.size();
	while(start_len < len){
		if(start_len + current_len > len){
			current_len = len - start_len  - 1;
		}
		if(isPalindrome(s, start_len, start_len + current_len)){
			cnt ++;
		}
		current_len ++;
		start_len += current_len;
		
	}
 	cout << cnt << endl;
	return 0;
}

int isPalindrome(string s, int start, int end){
	if(start > end){
		return 0;
	}
	while(start <= end){
		if(s[start] != s[end]){
			return 0;
		}
		start ++;
		end --;
	}
	return 1;
}
