```
#include<iostream>
#include<string>

using namespace std;

int main ()
{
	string s;
	cin >> s;
	int len = s.size();
	for(int i = 0; i < len; i++){
		if(s[i] == 'T'){
			s[i] = 'A';
		}
		else if(s[i] == 'A'){
			s[i] = 'T';
		}
		else if(s[i] == 'G'){
			s[i] = 'C';
		}
		else if(s[i] == 'C'){
			s[i] = 'G';
		}
	}
	cout << s << endl;
	return 0;
}
```
