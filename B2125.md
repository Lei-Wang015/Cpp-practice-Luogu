#include<bits/stdc++.h>
using namespace std;
struct Student{
	char name[23];
	int grade;
}st[114];

bool cmp(Student A, Student B){
	return A.grade > B.grade;
}

int main ()
{
	int n;
	cin >> n;
	for(int i = 0; i <= n; i++){
		cin >> st[i].grade;
		cin >> st[i].name;
	}
	sort(st + 1, st + n + 1, cmp);
	cout << st[1].name;
	return 0;
}
