```
#include<iostream>
#include<string>

using namespace std;

int main ()
{
	string s;
	getline(cin, s);
	int len = s.size();
	for(int i = 0;i < len; i++){
		if(s[i] > 'a' && s[i] <= 'z' || s[i] > 'A' && s[i] <= 'Z'){
			s[i] -= 1;
		}
		else if(s[i] == 'a'){
			s[i] = 'z';
		}
		else if(s[i] == 'A'){
			s[i] = 'Z';
		}
	}
	cout << s << endl;
	return 0;
}
```
